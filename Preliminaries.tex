\section{Preliminaries}
\color{red}
Definitions: 
\begin{enumerate}
    \item oligomorphic
    \item homogeneous
    \item smooth approximation by homogeneous substructures \cite{KLM89} (N.B. not the 'smooth approximation' from \cite[Definitino~4]{MP24})
    \item \emph{oligomorphic approximation} of a homogeneous structure by finite substructures with uniformly few orbits (i.e., types) that cover the age of $\A$
\end{enumerate}

\begin{theorem}
    Oligomorphically approximable homogeneous structures have the finite length property over any field of characteristic $0$.
\end{theorem}
\begin{proof}
    Copy the `bojań-trick' from \S8.2 of Mikołaj's
    \url{https://www.mimuw.edu.pl/~bojan/papers/notes-July3.pdf}.
\end{proof}
