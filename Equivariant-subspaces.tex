\section{All those equivariant subspaces}\label{sec:equivariant-subspaces}
We continue working in the setting of Section~\ref{sec:free-amalg} so that we can demonstrate an important application of Theorem~\ref{thm:cog-span-generally}:
given an orbit-finite set $X$ over $\A$, we can find a family of equivariant maps out of $\Lin_\FF X$, indexed by finite-dimensional vector spaces, whose kernels exhaust all the equivariant subspaces of $\Lin_\FF X$.
In particular, we will show that for a single $S$-ordered orbit $\cal O \subseteq \A^I$, the length of $\Lin_\FF \cal O$ is precisely $2^{|I|}$.


Let $X$ be an orbit-finite set over $\A$.
Then $X$ is isomorphic to a finite disjoint union $\biguplus_k \cal O_k$ of equivariant ordered orbits, where $\cal O_k \subseteq \A^{I_k}$.
For $J \subseteq I_k$, we have an equivariant map $\cal O_k \twoheadrightarrow \cal O_k|^J \hookrightarrow \A^{|J|}$.
The images of $\cal O_k|^J$ and $\cal O_{k'}|^{J'}$ agree only if $|J| = |J'|$.
Group $\cal O_k|^J$ by their images $\Jclass_i$.

Now, let $\mathcal{Q}_{1} = \mathcal{O}|^{J_1}, \dots, \mathcal{Q}_{t} = \mathcal{O}|^{J_t}$ be distinct projected $S$-ordered orbits up to $\Aut(\A/S)$-equivariant bijections,
enumerated in such a way that $|J_1| \geq |J_2| \geq \dots \geq |J_t|$. (In particular, $J_1=I$ and $J_t=\emptyset$.)

%\begin{definition}
    For $i = 1, \dots, t$, let $\Jclass_i$ consist of all sets $J$ such that $\mathcal{O}|^J$ is in $\Aut(\A/S)$-equivariant bijection with $\mathcal{Q}_i$.
    Assemble all $|\Jclass_i|$ projections into a single map
    \[
        (-){\restriction_i} : \Lin_\FF\mathcal{O} \to \Lin_{\FF^{\Jclass_i}} \mathcal{Q}_i.
    \]
    To be more precise $v{\restriction_i}(a)$ is, for $a\in{\cal Q}_i$, the $\Jclass_i$-tuple whose $J$-th entry is $v|^J(a^{/J}) \in \FF$.
    It is straightforward to check that $(-){\restriction_i}$ is $\Aut(\A/S)$-equivariant and linear.
%\end{definition}


\begin{theorem}
Every equivariant subspace of $\Lin X$ is of the form
\[
    \bigcap_{i} \ker( \Lin X \xrightarrow{ \restriction_i } \Lin_{\FF^{\Jclass_i}} \cal Q_i \xrightarrow{\text{proj}} \Lin_{V_i^\perp} \cal Q_i )
\]
where $V_i \subseteq \FF^{\Jclass_i}$ is a (finite-dimensional) subspace.
\end{theorem}

\begin{theorem}
    $\len(\Lin_\FF\mathcal{O}) = 2^{|I|}$.
\end{theorem}
